# Canvas Apis
We will discuss about canvas apis

## Clear
Fills clip with color, means the whole canvas(clipped zone, if applied) with the given color.

`void clear(SkColor color)`
`void clear(const SkColor4f& color)`

Use `SkColorSetRGB(r,g,b)` to create your color and use to clear the canvas.

Example:
```cpp
canvas->clear(SkColorSetRGB(244, 120, 0));
```

## DrawColor
If you want control on blend mode to clear the whole clipped area then you should use `drawColor`.

`void drawColor(SkColor color, SkBlendMode mode = SkBlendMode::kSrcOver)`

`void drawColor(const SkColor4f& color, SkBlendMode mode = SkBlendMode::kSrcOver)`

Example:
```cpp
canvas->drawColor(SkColorSetARGB(255, 244, 120, 0), SkBlendMode::kSrc);
canvas->drawColor(SkColorSetARGB(255, 0, 120, 0), SkBlendMode::kColorDodge);
```

## Paint

* color

## DrawImageNine

`void drawImageNine(const SkImage* image, const SkIRect& center, const SkRect& dst, SkFilterMode filter, const SkPaint* paint = nullptr);`

Example:
```cpp
canvas->clear(SK_ColorWHITE);

// --- load image
std::filesystem::path path = "C:\\Users\\deb\\Downloads\\rrect.png";
auto data = SkData::MakeFromFileName(path.string().c_str());
if (!data) return;

auto image = SkImages::DeferredFromEncodedData(data);
if (!image) return;

int w = image->width();
int h = image->height();

// --- IMPORTANT: define border thickness (must match your image corners)
// your image shows thick rounded border → estimate ~40px
int border = 20;

SkIRect center = SkIRect::MakeLTRB(
    border,
    border,
    w - border,
    h - border
);

// --- destination size (resize freely)
SkRect dst = SkRect::MakeXYWH(50, 50, 500, 250);

// --- draw 9-patch
canvas->drawImageNine(
    image.get(),
    center,
    dst,
    SkFilterMode::kLinear,
    nullptr
);
```

```cpp
    h / 5,
    2 * h / 5,
    3 * h / 5,
    4 * h / 5
};

SkCanvas::Lattice lattice{};
lattice.fXDivs = xDivs;
lattice.fYDivs = yDivs;
lattice.fXCount = 5;
lattice.fYCount = 4;

lattice.fRectTypes = nullptr;
lattice.fColors = nullptr;
lattice.fBounds = nullptr;

// BIG stretch area → exaggerated effect
SkRect dst = SkRect::MakeXYWH(20, 20, 900, 600);

canvas->drawImageLattice(
    image.get(),
    lattice,
    dst,
    SkFilterMode::kNearest, // 🔥 pixelated = more impact
    nullptr
);
```

## FontSystem

```cpp
#include <string>

#include "include/core/SkFontMgr.h"
#include "include/core/SkRefCnt.h"

        void Reset();

        sk_sp<SkFontMgr> GetFontManager() const;
        sk_sp<skia::textlayout::FontCollection> GetCollection() const;
        sk_sp<SkUnicode> GetUnicode() const;
        sk_sp<SkTypeface> MatchTypeface(const std::string& family, const SkFontStyle& style) const;
        SkString AddFromFile(const char path[], const char* name = nullptr);

    private:
```

```cpp
// FontSystem.cpp

#include "FontSystem.h"

#include "include/ports/SkTypeface_win.h"
#include "modules/skunicode/include/SkUnicode_icu.h"

namespace skiaexplorer {
    void FontSystem::Initialize()
    {
        EnsureInitialized();
    }

    void FontSystem::Reset()
    {
        fontProvider_.reset();
        fontCollection_.reset();
        unicode_.reset();
        fontManager_.reset();
    }

    sk_sp<SkFontMgr> FontSystem::GetFontManager() const
    {
        EnsureInitialized();
        return fontManager_;
    }

    sk_sp<skia::textlayout::FontCollection> FontSystem::GetCollection() const
    {
        EnsureInitialized();
        return fontCollection_;
    }

    sk_sp<SkUnicode> FontSystem::GetUnicode() const
    {
        EnsureInitialized();
        return unicode_;
    }

    sk_sp<SkTypeface> FontSystem::MatchTypeface(const std::string& family, const SkFontStyle& style) const
    {
        EnsureInitialized();
        if (!fontManager_)
        {
            return nullptr;
        }

        sk_sp<SkTypeface> typeface = fontManager_->legacyMakeTypeface(
            family.empty() ? nullptr : family.c_str(),
            style);
        if (!typeface && !family.empty())
        {
            typeface = fontManager_->legacyMakeTypeface(nullptr, style);
        }
        return typeface;
    }

    SkString FontSystem::AddFromFile(const char path[], const char* name)
    {
        EnsureInitialized();
        if (!fontManager_ || !fontProvider_)
        {
            return {};
        }

        sk_sp<SkTypeface> typeface = fontManager_->makeFromFile(path, 0);
        if (!typeface)
        {
            return {};
        }

        if (name)
        {
            SkString familyName(name);
            fontProvider_->registerTypeface(typeface, familyName);
            return familyName;
        }

        SkString familyName;
        typeface->getFamilyName(&familyName);
        fontProvider_->registerTypeface(typeface, familyName);
        return familyName;
    }

    void FontSystem::EnsureInitialized() const
    {
        auto* self = const_cast<FontSystem*>(this);
        if (!self->fontManager_)
        {
            self->fontManager_ = SkFontMgr_New_DirectWrite();
            if (!self->fontManager_)
            {
                self->fontManager_ = SkFontMgr_New_GDI();
            }
        }
        if (!self->fontProvider_)
        {
            self->fontProvider_ = sk_make_sp<skia::textlayout::TypefaceFontProvider>();
        }
        if (!self->fontCollection_)
        {
            self->fontCollection_ = sk_make_sp<skia::textlayout::FontCollection>();
            self->fontCollection_->setAssetFontManager(self->fontProvider_);
            self->fontCollection_->setDefaultFontManager(self->fontManager_);
        }
        if (!self->unicode_)
        {
            self->unicode_ = SkUnicodes::ICU::Make();
        }
    }
}
```
