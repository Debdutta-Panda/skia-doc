# Skia RuntimeEffect and SkSL Playground Example

`SkRuntimeEffect` is the fastest way to move from built-in Skia effects to custom shader logic.

This example shows a simple interactive-style shader scene with:

- one SkSL shader
- named uniform binding via `SkRuntimeShaderBuilder`
- animated parameters
- reuse across multiple shapes

## Example

```cpp
#include "include/core/SkCanvas.h"
#include "include/core/SkPaint.h"
#include "include/core/SkRect.h"
#include "include/core/SkShader.h"
#include "include/effects/SkRuntimeEffect.h"

static sk_sp<SkShader> MakePlaygroundShader(const SkRect& rect, float timeSeconds) {
    static sk_sp<SkRuntimeEffect> effect;
    static bool attemptedCompile = false;
    if (!attemptedCompile) {
        attemptedCompile = true;
        auto result = SkRuntimeEffect::MakeForShader(SkString(R"(
            uniform float2 resolution;
            uniform float time;
            uniform float4 accentA;
            uniform float4 accentB;

            float4 main(float2 xy) {
                float2 uv = xy / resolution;
                uv = uv * 2.0 - 1.0;

                float r = length(uv);
                float angle = atan(uv.y, uv.x);

                float pulse = 0.5 + 0.5 * sin(time * 2.0 + r * 12.0);
                float waves = 0.5 + 0.5 * cos(angle * 6.0 - time * 3.0);
                float glow = 1.0 - smoothstep(0.15, 0.95, r);

                float3 color = mix(accentA.rgb, accentB.rgb, waves);
                color += float3(0.18 * pulse, 0.07 * pulse, 0.12 * pulse);
                color = color * glow + float3(0.05, 0.05, 0.07);

                return float4(color, 1.0);
            }
        )"));
        if (!result.effect && !result.errorText.isEmpty()) {
            SkDebugf("RuntimeEffect compile failed:\\n%s\\n", result.errorText.c_str());
        }
        effect = std::move(result.effect);
    }

    if (!effect) {
        return SkShaders::Color(SkColorSetRGB(90, 20, 20));
    }

    SkRuntimeShaderBuilder builder(effect);
    builder.uniform("resolution") = SkV2{ rect.width(), rect.height() };
    builder.uniform("time") = timeSeconds;
    builder.uniform("accentA") = SkV4{ 0.98f, 0.40f, 0.28f, 1.0f };
    builder.uniform("accentB") = SkV4{ 0.20f, 0.54f, 0.92f, 1.0f };

    auto shader = builder.makeShader(nullptr);
    if (!shader) {
        return SkShaders::Color(SkColorSetRGB(20, 70, 120));
    }
    return shader;
}

void DrawScene(SkCanvas* canvas, const AppState& state) {
    canvas->clear(SkColorSetRGB(18, 20, 28));

    const float t = static_cast<float>(state.timeSeconds);
    const SkRect panel = SkRect::MakeXYWH(70, 60, 700, 420);

    auto shader = MakePlaygroundShader(panel, t);

    SkPaint p;
    p.setAntiAlias(true);
    p.setShader(shader);

    canvas->drawRoundRect(panel, 28, 28, p);

    SkPaint overlayCircle;
    overlayCircle.setAntiAlias(true);
    overlayCircle.setShader(MakePlaygroundShader(SkRect::MakeXYWH(0, 0, 180, 180), t + 0.7f));
    canvas->save();
    canvas->translate(30, 120);
    canvas->drawCircle(90, 90, 90, overlayCircle);
    canvas->restore();

    SkPaint overlayRect;
    overlayRect.setAntiAlias(true);
    overlayRect.setShader(MakePlaygroundShader(SkRect::MakeXYWH(0, 0, 140, 140), t + 1.1f));
    canvas->save();
    canvas->translate(600, 110);
    canvas->rotate(18, 70, 70);
    canvas->drawRect(SkRect::MakeXYWH(0, 0, 140, 140), overlayRect);
    canvas->restore();
}
```

## Why this example matters

- it shows one effect reused in multiple places
- it demonstrates safer named uniform binding
- it gives you a foundation for future animated backgrounds, procedural fills, and live tooling

## Practical next steps
